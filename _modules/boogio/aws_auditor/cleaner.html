[docs]class Cleaner(object):
    '''
    Remove timestamped files meeting specified criteria.
    '''

    _supported_periods = ['hour', 'day', 'week']

    def __init__(
            self,
            config,
            *default_periods
            ):
        '''
        Initialize a Cleaner instance.

        A cleaner config specifies how many periods back to preserve
        existing files. Any files with a timestamp from more periods
        back than the count for their period, except for the earliest
        in each period, will be deleted. The earliest in each period
        will be retained as an archive.

        Here's a simple cleaner config. This can be passed as a python
        structure or the name of a json file.
        {
        'hour': HOUR_COUNT,
        'day': DAY_COUNT,
        'week': WEEK_COUNT,
        }

        All files older than the start of the previous cleanup period
        will be deleted, unless their timestamp is the earliest in
        their cleanup period.

        For example, if the cleanup period is 'hour', and the current
        time is 0420, all files whose timestamp is earlier than 0300
        will be deleted except for the files whose timestamps are
        closest to the start of the hour in which they fall.

        If the timestamps are at 0216, 0230, 0245, 0255, 0315, 0330,
        0340, 0402 and 0414, an 0420 'hour' cleanup will remove the
        files with timestamps 0230, 0245 and 0255, leaving 0216, 0315,
        0330, 0340, 0402 and 0414. Any additional cleanups before 0500
        won't remove any more files. A subsequent 0500 cleanup will
        remove 0330 and 0340, leaving 0216, 0315, 0402 and 0414.

        Summarizing, if the file timestamps in each hour are as
        follows:

            02xx:   0216, 0230, 0245, 0255
            03xx:   0315, 0330, 0340
            04xx:   0402, 0414

        an 'hour' cleanup in the 04xx hour window will remove all but
        the earliest file in the 02xx window and any earlier window.


        '''
        unsupported_periods = [
            p for p in default_periods
            if p not in self._supported_periods
            ]
        if len(unsupported_periods) > 0:
            raise ValueError(
                'unsupported periods: %s ' % ', '.join(unsupported_periods)
                )

        self.default_periods = list(default_periods)

        # Treat config as a filename pointing to a json config file. If
        # that doesn't work, assume it's the actual config dict.
        try:
            with open(config) as fp:
                self.config = json.load(fp)
        except TypeError:
            self.config = config
