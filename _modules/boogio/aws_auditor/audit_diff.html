[docs]def index_by_keys(audit, keys, require_unique=True):
    '''
    Return a dict indexing an audit report by a list of keys, and a
    list of all keys found in any entry in the audit report.

    The value for each item in the index returned is a list of all
    items in the audit report with values for the key fields matching
    the values used to construct the key for that index item.

    If require_unique is True and index_by_keys doesn't raise an
    exception, the values list for each item in the index returned
    will be of length one.

    '''
    index = {}
    indexed_records = 0
    keys_found = {}

    for record in audit:
        keys_found.update({k: True for k in record.keys()})
        index_key = str([record[key] for key in keys])

        if index_key not in index:
            index[index_key] = []

        else:
            if require_unique:
                raise DuplicateRecordError(index_key)

        index[index_key].append(record)
        indexed_records += 1

    assert(indexed_records == len(audit))

    return (index, keys_found.keys())

def _load_json_if_file_exists(filename):
    '''
    If a string is the path to a file, try to load JSON data and return
    the resulting structure. If string isn't the path to a file, return
    the string.

    If any error other than file non-existence is encountered, raise it.
    '''
    try:
        with open(filename, 'r') as fp:
            return_val = json.load(fp)
    except IOError as err:
        if err.errno == 2:
            # file doesn't exist
            return_val = filename
        else:
            raise err

    return return_val

[docs]    def compare(
            self,
            audit0,
            audit1=None,
            uid=None,
            require_unique=True,
            ignore=[],
            ):
        '''
        Calculate differences between two audit reports.

        Parameters:

        audit0, audit1 -- The two audit reports to compare. These should
            be lists of dicts, and each dict should be flat in the sense
            of the flatten package. These may also be the paths to json
            files defining such dicts.

            If audit1 is None, self.baseline will be used for audit0 and
            the audit report passed as audit0 will be used for audit1.

        uid -- A list of keys that should be considered to identify an
            audit record as representing a unique entity.

        require_unique -- Whether multiple records with the same uid
            values are allowed. If set to True, an exception will be
            raised if more than one record with the same values for the
            keys in uid occurs in the same audit report.

        ignore -- A list of keys, differences in whose values won't be
            considered in determining if records have changed.

        Records will be "expected" the same if they have the same
        values for the keys listed in uid. Differences in the values
        for other keys will be reported. Any uid value combinations
        not appearing in both audits will be reported as additions or
        deletions depending on the audit in which they appear.

        Return Value:

        compare() returns a dict with three elements:
            {
            'uid': uid,
            'key_diff': key_diff,
            'record_diff': record_diff
            }

        uid -- the uid passed in to compare().

        key_diff -- a pair [audit0_uniq_keys, audit1_uniq_keys], where
            each element is a list of the keys that occur in the indicated
            audit but not the other.

        record_diff -- a list of all pairs [audit0_item, audit1_item] where
            audit0_item and audit1_item have the same values for all of the
            keys in uid but differ in some other value. Values for keys in
            key_diff aren't considered when calculating record_diff, but
            audit items differeing in their common keys will be included.
        '''
        # Try to load from JSON.
        if isinstance(audit0, str):
            audit0 = _load_json_if_file_exists(audit0)
        if isinstance(audit1, str):
            audit1 = _load_json_if_file_exists(audit1)

        # If only one audit is passed, we compare it to this audit diff's
        # baseline audit.
        if audit1 is None:
            audit1 = audit0
            audit0 = self.baseline

        # Index items according to their values for the keys in uid.
        (index0, audit0_keys) = index_by_keys(audit0, uid, require_unique=True)
        (index1, audit1_keys) = index_by_keys(audit1, uid, require_unique=True)

        # Keys that only appear in one or the other audit.
        audit0_uniq_keys = [k for k in audit0_keys if k not in audit1_keys]
        audit1_uniq_keys = [k for k in audit1_keys if k not in audit0_keys]

        # # Keys that appear in some record in each audit.
        # audit_common_keys = [k for k in audit0_keys if k in audit1_keys]

        # Value combinations for UID keys that appear in one or the other or
        # both audits. The values in the indices are of length one because
        # we called index_by_keys() with require_unique=True.
        index0_only = {k: v[0] for (k, v) in index0.items() if k not in index1}
        index0_shared = {k: v[0] for (k, v) in index0.items() if k in index1}
        index1_only = {k: v[0] for (k, v) in index1.items() if k not in index0}
        index1_shared = {k: v[0] for (k, v) in index1.items() if k in index0}

        # The specific records differ but there better be the same number.
        assert(len(index0_shared) == len(index1_shared))

        # Our return values, eventually.
        key_diff = [audit0_uniq_keys, audit1_uniq_keys]
        record_diff = []

        # Records for entities that only appear in the first audit.
        if len(index0_only) > 0:
            record_diff = record_diff + [
                [v, None] for v in index0_only.values()
                ]

        # Records for entities that only appear in the second audit.
        if len(index1_only) > 0:
            record_diff = record_diff + [
                [None, v] for v in index1_only.values()
                ]

        # Records for entities that appear in both audits. We treat them
        # as part of the diff if they differ in any values, or if they
        # have any unshared keys that aren't in the overall unique keys.
        # The latter are omitted here because every record would show up
        # as different.
        #
        # TODO: Add an option to include audit*_uniq_keys differences.
        # If the option is set, use [] as the audit*_key_ignore list.
        audit0_key_ignore = audit0_uniq_keys
        audit1_key_ignore = audit1_uniq_keys

        for (common_index, record_0) in index0_shared.items():

            record_1 = index1_shared[common_index]
            # record0 and record1 are audit records with common uid values.

            unshared_keys_0 = [
                k for k in record_0
                if k not in record_1 and k not in audit0_key_ignore
                ]
            unshared_keys_1 = [
                k for k in record_1
                if k not in record_0 and k not in audit1_key_ignore
                ]

            # print record_0
            # print record_1
            value_diff_keys = [
                k for k in record_0
                if (
                    k in record_1
                    and k not in ignore
                    and record_0[k] != record_1[k]
                    )
                ]

            if max([
                    len(value_diff_keys),
                    len(unshared_keys_0),
                    len(unshared_keys_1)
                    ]) > 0:
                record_diff.append([record_0, record_1])

        return {
            'uid': uid,
            'key_diff': key_diff,
            'record_diff': record_diff
            }

[docs]    def added(
            self,
            audit0,
            audit1=None,
            uid=None,
            require_unique=True,
            ignore=[],
            ):
        '''
        Find records in audit1 that aren't in audit0.
        '''
        comparison = self.compare(
            audit0,
            audit1=audit1,
            uid=uid,
            require_unique=require_unique,
            ignore=ignore
            )

        return [
            rd[1] for rd in comparison['record_diff']
            if rd[0] is None
            ]

[docs]    def changed(
            self,
            audit0,
            audit1=None,
            uid=None,
            require_unique=True,
            ignore=[],
            ):
        '''
        Find records in both audit0 and audit1 that changed.
        '''
        comparison = self.compare(
            audit0,
            audit1=audit1,
            uid=uid,
            require_unique=require_unique,
            ignore=ignore
            )

        return [
            rd for rd in comparison['record_diff']
            if None not in [rd[0], rd[1]]
            ]
