        params.update(kwargs)

[docs]    def add_dataset(self, dset, validation=True):
        '''
        Add a dataset to the _dispatch structure. The dataset can be
        given as a python structure, the path to a JSON file or a DataSet
        instance.

        Unless the validation flag is explicitly set to False, a
        ValidationError will be raised if the dataset isn't schema
        compliant. If this occurs, the calling DataDispatch will not
        be modified.
        '''

        if isinstance(dset, dataset.DataSet):
            new_ds = dset.to_dict()

        elif isinstance(dset, str):
            with open(dset, 'r') as fp:
                new_ds = json.load(fp)

        else:
            new_ds = copy.deepcopy(dset)

        # new_ds should now be a python dict representation of a dataset's
        # _dispatch content.

        if validation:
            validate(new_ds, 'dataset')

        self._dispatch['datasets'].append(new_ds)
