'''
Generate ElasticSearch index entries to interleave with strings of JSON
output.

Initialize an ElasticSearchIndexer instance with an INDEX_NAME and TYPE_NAME,
and optionally a starting ID_VALUE, as shown in the usage examples that
follow. Then call next_index_json each time a new ElasticSearch index is
needed, typically before each element of a list of json strings is to be
printed.

Usage:

    >>> import elastic_search_indexer

    >>> indexer = ElasticSearchIndexer(
    ...     {'_index': 'INDEX_NAME', '_type': 'TYPE_NAME'}
    ...     )
    >>> for json_output in ['{"thing0"}', '{"thing1"}']
    ...     print indexer.next_index_json()
    ...     print json_output

Resulting output:

    {"index":{"_index": "INDEX_NAME", "_type": "TYPE_NAME", "_id": 0}}
    {"thing0"}
    {"index":{"_index": "INDEX_NAME", "_type": "TYPE_NAME", "_id": 1}}
    {"thing1"}


Usage specifying a starting index:

    >>> import elastic_search_indexer

    >>> indexer = ElasticSearchIndexer(
    ...     {'_index': 'INDEX_NAME', '_type': 'TYPE_NAME', '_id': 42}
    ...     )
    >>> for json_output in ['{"thing0"}', '{"thing1"}', '{"thing2"}']
    ...     print indexer.next_index_json()
    ...     print json_output

Resulting output:

    {"index":{"_index": "INDEX_NAME", "_type": "TYPE_NAME", "_id": 42}}
    {"thing0"}
    {"index":{"_index": "INDEX_NAME", "_type": "TYPE_NAME", "_id": 43}}
    {"thing1"}
    {"index":{"_index": "INDEX_NAME", "_type": "TYPE_NAME", "_id": 44}}
    {"thing3"}

'''

    # - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
    def __init__(self, index_spec):
        '''
        Initialize an ElasticSearchIndexer instance.

        The index_spec parameter specifies how to populate the fields
        of each index created.

        The index_spec is a dict with the following structure:
        {
            '_index': 'INDEX_VALUE',
            '_type': 'TYPE_VALUE',
            '_id': ID_VALUE,
            }

        For future support for pulling the values for the index fields from
        the structures being indexed, the following structure is also
        supported. This is how the index_spec is represented internally; if
        the simpler structure is used for initialization it is convereted.
        {
            '_index': {'source': 'INDEX_SOURCE', 'value': 'INDEX_VALUE'},
            '_type': {'source': 'TYPE_SOURCE', 'value': 'TYPE_VALUE'},
            '_id': {'source': 'ID_SOURCE', 'value': ID_VALUE},
            }

        If 'INDEX_SOURCE' or 'TYPE_SOURCE' is None or the 'source' key is
        missing, the corresponding 'INDEX_VALUE' or 'TYPE_VALUE' will be used
        as the value for the '_index' and '_type' fields of the generated
        index.

        If ID_SOURCE is None, the '_id's for each index generated will be
        successive integers starting from the integer ID_VALUE.

        Additional options for *_SOURCE are not currently implemented.
        '''

        # These are the types of the 'value' field in the "full internal"
        # structure for the index_spec. In the "abbreviated" format these
        # are the types that must be provided for each _key.
        _index_value_schema = {
            '_index': str,
            '_type': str,
            '_id': int
            }

        # _id is optional, and defaults to zero.
        if '_id' not in index_spec:
            index_spec['_id'] = {'source': None, 'value': 0}

        # Convert "abbreviated" syntax to "full internal" structure.
        for key in ['_index', '_type', '_id']:
            if key in ['_index', '_type'] and key not in index_spec:
                # _index and _type are required.
                raise KeyError(
                    '_index and _type fields are required when initializing'
                    ' an ElasticSearchIndexer instance'
                    )

            if isinstance(index_spec[key], _index_value_schema[key]):
                index_spec[key] = {
                    'source': None, 'value': index_spec[key]
                    }

        self.index_spec = index_spec

        self.count = 0

    # - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
    def _index(self, json_data=None):
        '''
        Construct the _index, using the json_data provided if required by
        the index_spec.

        json_data is accepted as a parameter for future support of extracting
        the index fields from the entity being indexed.
        '''
        return self.index_spec['_index']['value']

    # - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
    def _type(self, json_data=None):
        '''
        Construct the _type, using the json_data provided if required by
        the index_spec.

        json_data is accepted as a parameter for future support of extracting
        the index fields from the entity being indexed.
        '''
        return self.index_spec['_type']['value']

    # - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
    def _incremented_id(self, json_data=None):
        '''
        Construct the _id, using the json_data provided if required by
        the index_spec.

        json_data is accepted as a parameter for future support of extracting
        the index fields from the entity being indexed.
        '''
        return self.index_spec['_id']['value'] + self.count

[docs]    def next_index(self, json_data=None):
        '''
        Return the next index as a python structure.

        json_data is accepted as a parameter for future support of extracting
        the index fields from the entity being indexed.
        '''
        index = {'index': {}}
        index['index']['_index'] = self._index(json_data)
        index['index']['_type'] = self._type(json_data)
        index['index']['_id'] = self._incremented_id(json_data)

        self.count = self.count + 1

        return index

[docs]    def next_index_json(self, json_data=None):
        '''
        Return the next index as a json string.

        json_data is accepted as a parameter for future support of extracting
        the index fields from the entity being indexed.
        '''
        return json.dumps(self.next_index(json_data))
