[docs]def wrap(this_thing, **kwargs):
    '''

    Add fields as specified in kwargs to the structure in this_thing.
    If this_thing already has meta and data fields, update the meta
    fields; otherwise, create them and add this_thing as the value of
    the data field.

    '''

    wrapped = {}
    meta = {}

    try:
        if 'meta' in this_thing:
            meta = this_thing['meta']
    except TypeError:
        pass

    meta.update(kwargs)
    wrapped['meta'] = meta

    try:
        if 'data' in this_thing:
            wrapped['data'] = this_thing['data']
        else:
            wrapped['data'] = this_thing

    except TypeError:
        wrapped['data'] = this_thing

    return wrapped

[docs]def json_load_wrap(this_thing, filename):
    '''
    Read in fields from a JSON file and wrap them around this_thing.
    '''

    with open(filename, 'r') as fp:
        meta = json.load(fp)

    return wrap(this_thing, **meta)
